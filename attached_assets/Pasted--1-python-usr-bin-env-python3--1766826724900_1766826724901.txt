import asyncio
import pandas as pd
from telethon import TelegramClient
from telethon.tl.functions.messages import GetHistoryRequest

async def rapid_chat_dump(chat_id, limit=500):
    """–®–≤–∏–¥–∫–∏–π –¥–∞–º–ø –≤—Å—ñ—Ö –¥–∞–Ω–∏—Ö –∑ —á–∞—Ç—É"""

        entity = await client.get_entity(chat_id)

        all_users = {}
        
        offset_id = 0
        total = 0
        
        print(f"üóÇÔ∏è –ü–æ—á–∞—Ç–æ–∫ –∑–±–æ—Ä—É –¥–∞–Ω–∏—Ö –∑ {chat_id}...")

                peer=entity,
                offset_id=offset_id,
                offset_date=None,
                add_offset=0,
                limit=100,
                max_id=0,
                min_id=0,
                hash=0
            ))
            
            if not history.messages:
                break
            
            messages = history.messages
            offset_id = messages[-1].id
            
            for msg in messages:
                # –ó–±—ñ—Ä —ñ–Ω—Ñ–æ—Ä–º–∞—Ü—ñ—ó –ø—Ä–æ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è
                message_data = {
                    'id': msg.id,
                    'date': msg.date,
                    'from_id': msg.from_id,
                    'text': msg.message or '',
                    'media': bool(msg.media),
                }
                all_messages.append(message_data)
                
                # –ó–±—ñ—Ä —ñ–Ω—Ñ–æ—Ä–º–∞—Ü—ñ—ó –ø—Ä–æ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á—ñ–≤
                if msg.from_id and msg.from_id not in all_users:
                    try:
                        user = await client.get_entity(msg.from_id)
                        all_users[msg.from_id] = {
                            'id': user.id,
                            'username': user.username,
                            'first_name': user.first_name,
                            'last_name': user.last_name,
                            'phone': getattr(user, 'phone', None),
                            'bot': user.bot,
                        }
                    except:
                        all_users[msg.from_id] = {'id': msg.from_id, 'error': '–ù–µ –≤–¥–∞–ª–æ—Å—è –æ—Ç—Ä–∏–º–∞—Ç–∏'}
            
            total += len(messages)
            print(f"üì• –ó—ñ–±—Ä–∞–Ω–æ {total} –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω—å...")
            
            if len(messages) < 100:
                break
        
        await client.disconnect()
        
        # –°—Ç–≤–æ—Ä–µ–Ω–Ω—è DataFrame
        messages_df = pd.DataFrame(all_messages)
        users_df = pd.DataFrame(all_users.values())
        
        # –ó–±–µ—Ä–µ–∂–µ–Ω–Ω—è –≤ Excel
        with pd.ExcelWriter(f'chat_dump_{chat_id}_{datetime.now().strftime("%H%M")}.xlsx') as writer:
            messages_df.to_excel(writer, sheet_name='–ü–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è', index=False)
            users_df.to_excel(writer, sheet_name='–ö–æ—Ä–∏—Å—Ç—É–≤–∞—á—ñ', index=False)
        
        print(f"‚úÖ –ó—ñ–±—Ä–∞–Ω–æ {total} –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω—å –≤—ñ–¥ {len(all_users)} –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á—ñ–≤")
        print(f"üíæ –ó–±–µ—Ä–µ–∂–µ–Ω–æ –≤ Excel —Ñ–∞–π–ª")
        
        return messages_df, users_df
        
    except Exception as e:
        print(f"‚ùå –ü–æ–º–∏–ª–∫–∞: {e}")
        return None, None
