    async def frontline_image_analysis(self, image_path: str, court_order: str):
        """–ï–∫—Å—Ç—Ä–µ–Ω–∏–π –∞–Ω–∞–ª—ñ–∑ –∑–æ–±—Ä–∞–∂–µ–Ω—å –Ω–∞ –º—ñ—Å—Ü—ñ"""
        try:
            from PIL import Image
            import hashlib
            
            # –ë–∞–∑–æ–≤–∏–π –∞–Ω–∞–ª—ñ–∑
            with Image.open(image_path) as img:
                # EXIF –¥–∞–Ω—ñ
                exif = img._getexif() or {}
                
                # –ì–µ–æ–ª–æ–∫–∞—Ü—ñ—è –∑ EXIF
                gps_data = self._extract_gps(exif)
                
                # –•–µ—à –¥–ª—è —ñ–¥–µ–Ω—Ç–∏—Ñ—ñ–∫–∞—Ü—ñ—ó
                with open(image_path, 'rb') as f:
                    file_hash = hashlib.sha256(f.read()).hexdigest()
                
                # –ú–µ—Ç–∞–¥–∞–Ω—ñ
                metadata = {
                    'filename': os.path.basename(image_path),
                    'size': os.path.getsize(image_path),
                    'dimensions': img.size,
                    'format': img.format,
                    'hash': file_hash,
                    'gps': gps_data,
                    'exif': {str(k): str(v) for k, v in exif.items()}
                }
                
                # –®–≤–∏–¥–∫–∞ –æ–±—Ä–æ–±–∫–∞ —á–µ—Ä–µ–∑ –º–æ–±—ñ–ª—å–Ω–∏–π ML
                detected_faces = await self._detect_faces_mobile(img)
                
                # –ü–æ—à—É–∫ –æ–±–ª–∏—á —É –±–∞–∑—ñ –¥–∞–Ω–∏—Ö
                matches = []
                for face in detected_faces:
                    match = await self._search_face_in_db(face)
                    if match:
                        matches.append(match)
                
                # –ó–±–µ—Ä–µ–∂–µ–Ω–Ω—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ–≤
                result = {
                    'operation_code': self.operation_code,
                    'officer_id': self.officer_id,
                    'court_order': court_order,
                    'image_analysis': {
                        'metadata': metadata,
                        'faces_detected': len(detected_faces),
                        'db_matches': matches,
                        'gps_location': gps_data
                    },
                    'timestamp': datetime.now().isoformat()
                }
                
                await self._save_evidence(
                    self._encrypt_data(result),
                    f"image_{file_hash}"
                )
                
                return result
                
        except Exception as e:
            self.logger.error(f"Image analysis failed: {e}")
            return {'status': 'error', 'message': str(e)}

    def _encrypt_data(self, data: dict) -> str:
        """–ü—Ä–æ—Å—Ç–µ —à–∏—Ñ—Ä—É–≤–∞–Ω–Ω—è AES –¥–ª—è –º–æ–±—ñ–ª—å–Ω–æ–≥–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è"""
        try:
            from Crypto.Cipher import AES
            from Crypto.Util.Padding import pad
            import base64
            
            data_str = json.dumps(data)
            cipher = AES.new(self.encryption_key[:32].encode(), AES.MODE_CBC)
            ct_bytes = cipher.encrypt(pad(data_str.encode(), AES.block_size))
            iv = base64.b64encode(cipher.iv).decode('utf-8')
            ct = base64.b64encode(ct_bytes).decode('utf-8')
            return json.dumps({'iv': iv, 'ciphertext': ct})
        except:
            # –†–µ–∑–µ—Ä–≤–Ω–∏–π –≤–∞—Ä—ñ–∞–Ω—Ç
            import base64
            return base64.b64encode(json.dumps(data).encode()).decode()
    
    async def _save_evidence(self, encrypted_data: str, filename: str):
        """–ó–±–µ—Ä–µ–∂–µ–Ω–Ω—è –¥–æ–∫–∞–∑—ñ–≤ –∑ GPS –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞–º–∏"""

        Path(evidence_dir).mkdir(parents=True, exist_ok=True)
        
        evidence_file = f"{evidence_dir}/{filename}.enc"
        with open(evidence_file, 'w', encoding='utf-8') as f:
            f.write(encrypted_data)
        
        # –î–æ–¥–∞–≤–∞–Ω–Ω—è GPS –º–µ—Ç–∫–∏
        gps_file = f"{evidence_dir}/{filename}_gps.txt"
        location = self._get_location()
        with open(gps_file, 'w') as f:
            f.write(json.dumps(location))
        
        self.logger.info(f"Evidence saved: {evidence_file}")
    
    def _get_location(self):
        """–û—Ç—Ä–∏–º–∞–Ω–Ω—è GPS –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç"""
        try:
            import gpsd
            gpsd.connect()
            packet = gpsd.get_current()
            return {
                'latitude': packet.lat,
                'longitude': packet.lon,
                'altitude': packet.alt,
                'speed': packet.hspeed,
                'time': packet.time
            }
        except:
            # –†–µ–∑–µ—Ä–≤–Ω–∏–π –≤–∞—Ä—ñ–∞–Ω—Ç —á–µ—Ä–µ–∑ Android
            try:
                import android.location
                location = android.location.LocationManager()
                return {
                    'latitude': location.getLastKnownLocation('gps').getLatitude(),
                    'longitude': location.getLastKnownLocation('gps').getLongitude(),
                    'provider': 'android'
                }
            except:
                return {'latitude': 0, 'longitude': 0, 'error': 'GPS unavailable'}

    async def _detect_faces_mobile(self, img):
        """–î–µ—Ç–µ–∫—Ü—ñ—è –æ–±–ª–∏—á —á–µ—Ä–µ–∑ –º–æ–±—ñ–ª—å–Ω–∏–π ML"""
        # –í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è TensorFlow Lite –¥–ª—è –º–æ–±—ñ–ª—å–Ω–∏—Ö –ø—Ä–∏—Å—Ç—Ä–æ—ó–≤
        try:
            import tflite_runtime.interpreter as tflite
            
            # –ó–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è –º–æ–¥–µ–ª—ñ –¥–µ—Ç–µ–∫—Ü—ñ—ó –æ–±–ª–∏—á
            model_path = "/sdcard/cyberpolice/models/face_detection.tflite"
            interpreter = tflite.Interpreter(model_path=model_path)
            interpreter.allocate_tensors()
            
            # –û–±—Ä–æ–±–∫–∞ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
            input_details = interpreter.get_input_details()
            output_details = interpreter.get_output_details()
            
            # –ú–∞—Å—à—Ç–∞–±—É–≤–∞–Ω–Ω—è –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
            input_shape = input_details[0]['shape']
            img_resized = img.resize((input_shape[1], input_shape[2]))
            
            # –í–∏–∫–æ–Ω–∞–Ω–Ω—è —ñ–Ω—Ñ–µ—Ä–µ–Ω—Å—É
            interpreter.set_tensor(input_details[0]['index'], [img_resized])
            interpreter.invoke()
            
            # –û—Ç—Ä–∏–º–∞–Ω–Ω—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ–≤
            faces = interpreter.get_tensor(output_details[0]['index'])
            
            return faces
            
        except Exception as e:
            self.logger.error(f"Face detection failed: {e}")
            return []
    
    async def _search_face_in_db(self, face_embedding):
        """–ü–æ—à—É–∫ –æ–±–ª–∏—á —É –ª–æ–∫–∞–ª—å–Ω—ñ–π –±–∞–∑—ñ –¥–∞–Ω–∏—Ö"""
        db_path = "/sdcard/cyberpolice/db/suspects.db"
        try:
            import sqlite3
            conn = sqlite3.connect(db_path)
            cursor = conn.cursor()
            
            # –ü–æ—à—É–∫ —Å—Ö–æ–∂–∏—Ö –æ–±–ª–∏—á (—Å–ø—Ä–æ—â–µ–Ω–∞ –≤–µ—Ä—Å—ñ—è)
            cursor.execute("""
                SELECT name, description, match_confidence 
                FROM suspects 
                ORDER BY embedding_distance(?) 
                LIMIT 3
            """, (json.dumps(face_embedding.tolist()),))
            
            results = cursor.fetchall()
            conn.close()
            
            return [{'name': r[0], 'description': r[1], 'confidence': r[2]} 
                   for r in results]
            
        except:
            return []
    
    def _check_local_db(self, phone_number):
        """–ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ –≤ –ª–æ–∫–∞–ª—å–Ω—ñ–π –±–∞–∑—ñ –≤—ñ–¥–æ–º–∏—Ö –Ω–æ–º–µ—Ä—ñ–≤"""
        db_path = "/sdcard/cyberpolice/db/phone_numbers.db"
        try:
            import sqlite3
            conn = sqlite3.connect(db_path)
            cursor = conn.cursor()
            
            cursor.execute("""
                SELECT category, description, first_seen 
                FROM known_numbers 
                WHERE phone_number = ? 
                LIMIT 1
            """, (phone_number,))
            
            result = cursor.fetchone()
            conn.close()
            
            if result:
                return {
                    'source': 'local_db',
                    'category': result[0],
                    'description': result[1],
                    'first_seen': result[2]
                }
        except:
            pass

class BattlefieldCrypto:
    """–ö—Ä–∏–ø—Ç–æ–≥—Ä–∞—Ñ—ñ—è –¥–ª—è —Ä–æ–±–æ—Ç–∏ –≤ —É–º–æ–≤–∞—Ö –±–æ–π–æ–≤–∏—Ö –¥—ñ–π"""

    def emergency_wipe(path: str, passes: int = 7):
        """–ï–∫—Å—Ç—Ä–µ–Ω–µ –∑–Ω–∏—â–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö –∑–∞ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–æ–º DoD 5220.22-M"""

        os.remove(path)
    
    @staticmethod
    def hide_data_in_image(image_path: str, data: bytes, output_path: str):
        """–ü—Ä–∏—Ö–æ–≤—É–≤–∞–Ω–Ω—è –¥–∞–Ω–∏—Ö –≤ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—ñ (—Å—Ç–µ–≥–∞–Ω–æ–≥—Ä–∞—Ñ—ñ—è)"""

        binary_data = ''.join(format(byte, '08b') for byte in data)
        
        # –ü—Ä–∏—Ö–æ–≤—É–≤–∞–Ω–Ω—è –≤ LSB (Least Significant Bit)
        pixels = list(img.getdata())

        new_pixels = []
        for pixel in pixels:
            if data_index < len(binary_data):
                # –ó–º—ñ–Ω–∞ –Ω–∞–π–º–µ–Ω—à –∑–Ω–∞—á—É—â–æ–≥–æ –±—ñ—Ç–∞
                new_pixel = list(pixel)

                    if data_index < len(binary_data):
                        new_pixel[i] = (pixel[i] & ~1) | int(binary_data[data_index])
                        data_index += 1
                new_pixels.append(tuple(new_pixel))

                new_pixels.append(pixel)
